end -}} {{- $context := dict "title" $title "description" $description "canonicalURL" $canonicalURL "ogType" $ogType "author" $author "publishedTime" $publishedTime "modifiedTime" $modifiedTime "robots" $robots "keywords" $keywords -}} {{- $context | jsonify -}}
