$seo.ogImageURL -}} {{- $schema = merge $schema (dict "image" $seo.ogImageURL) -}} {{- end -}} {{- else if $page.IsPage -}} {{- $schema = merge $schema (dict "@type" "BlogPosting" "headline" $schema (dict "description" $seo.description) -}} {{- end -}} {{- if $seo.ogImageURL -}} {{- $schema = merge $schema (dict "image" $seo.ogImageURL) -}} {{- end -}} {{- if $seo.author -}} {{- $schema = gt (len $page.Pages) 0 -}} {{- $schema = merge $schema (dict "numberOfItems" (len $page.Pages)) -}} {{- end -}} {{- end -}}
